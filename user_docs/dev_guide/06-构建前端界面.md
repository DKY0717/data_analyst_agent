> 本章目标：用 Vue 3 构建用户界面，让用户可以通过网页提问并查看结果。

## 6.3 配置 Vite 代理

> 前端运行在 `localhost:3000`，后端在 `localhost:8000`。浏览器会阻止跨域请求。
>
> **替换** `frontend/vite.config.js` 的全部内容为：

```javascript
import { fileURLToPath, URL } from 'node:url'
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'

export default defineConfig({
  plugins: [vue()],
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url)),
    },
  },
  server: {
    port: 3000,
    proxy: {
      '/api': {
        target: 'http://localhost:8000',
        changeOrigin: true,
      },
    },
  },
})
```

> **代理的作用：** 浏览器请求 `localhost:3000/api/chat/query` 时，Vite 会自动转发到 `localhost:8000/api/chat/query`，绕过跨域限制。

## 6.5 封装 API 调用

> 先创建需要的目录：

```bash
mkdir frontend/src/api
mkdir frontend/src/stores
mkdir frontend/src/components
```

> 创建 `frontend/src/api/agent.js`：

```javascript
import axios from 'axios'

const api = axios.create({
  baseURL: '/api',
  timeout: 60000,
})

export async function queryQuestion(question, sessionId) {
  const response = await api.post('/chat/query', {
    question,
    session_id: sessionId,
  })
  return response.data
}

export async function getSchema() {
  const response = await api.get('/schema')
  return response.data
}
```

## 6.6 创建状态管理

> 创建 `frontend/src/stores/query.js`：

```javascript
import { defineStore } from 'pinia'
import { queryQuestion } from '@/api/agent'

export const useQueryStore = defineStore('query', {
  state: () => ({
    question: '',
    loading: false,
    result: null,
    error: null,
    sessionId: null,
    history: [],
  }),

  actions: {
    async submitQuestion(question) {
      this.loading = true
      this.error = null
      this.question = question

      try {
        const response = await queryQuestion(question, this.sessionId)
        this.result = response.data
        this.sessionId = response.data.session_id
        this.history.push({
          question,
          answer: response.data.answer,
          time: new Date().toLocaleTimeString(),
        })
      } catch (err) {
        this.error = err.response?.data?.detail || err.message
      } finally {
        this.loading = false
      }
    },

    clearResult() {
      this.result = null
      this.error = null
    },
  },
})
```

## 6.7 创建核心组件

### QueryInput 组件

> 创建 `frontend/src/components/QueryInput.vue`：

```vue
<template>
  <div class="query-input">
    <el-input
      v-model="question"
      size="large"
      placeholder="请输入数据问题，例如：查询销售额最高的前5个商品"
      @keyup.enter="handleSubmit"
      clearable
    >
      <template #append>
        <el-button
          type="primary"
          :loading="store.loading"
          @click="handleSubmit"
        >
          查询
        </el-button>
      </template>
    </el-input>
  </div>
</template>

<script setup>
import { ref } from 'vue'
import { useQueryStore } from '@/stores/query'

const store = useQueryStore()
const question = ref('')

function handleSubmit() {
  if (question.value.trim() && !store.loading) {
    store.submitQuestion(question.value.trim())
  }
}
</script>

<style scoped>
.query-input {
  max-width: 800px;
  margin: 0 auto 24px;
}
</style>
```

### ResultTable 组件

> 创建 `frontend/src/components/ResultTable.vue`：

```vue
<template>
  <el-card v-if="store.result && store.result.rows?.length" class="table-card">
    <template #header>
      <span>查询结果（{{ store.result.rows.length }} 条）</span>
      <span class="time">耗时 {{ store.result.execution_time_ms }}ms</span>
    </template>
    <el-table :data="tableData" border stripe max-height="400">
      <el-table-column
        v-for="col in store.result.columns"
        :key="col"
        :prop="col"
        :label="col"
        sortable
      />
    </el-table>
  </el-card>
</template>

<script setup>
import { computed } from 'vue'
import { useQueryStore } from '@/stores/query'

const store = useQueryStore()

const tableData = computed(() => {
  if (!store.result?.rows) return []
  return store.result.rows.map(row => {
    const obj = {}
    store.result.columns.forEach((col, i) => {
      obj[col] = row[i]
    })
    return obj
  })
})
</script>

<style scoped>
.time {
  float: right;
  color: #999;
  font-size: 13px;
}
</style>
```

## 6.9 运行前端

