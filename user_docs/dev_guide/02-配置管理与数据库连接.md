> **开始前确认：** 你当前应该在项目根目录 `data_analyst_agent/` 下。如果不确定，在终端执行 `pwd` 查看路径。如果在 `backend/` 目录下，先执行 `cd ..` 回到根目录。

## 2.1 为什么需要配置管理

> 你的程序需要一些"变量"：
> - 数据库文件放在哪
> - API 监听哪个端口
> - 以后要接入的 LLM API Key
>
> 这些信息不应该写死在代码里，而是放在配置文件中。这样换环境（开发/测试/生产）时只需要改配置，不用改代码。

## 2.3 编写配置模块

> 创建 `backend/app/config.py`：

```python
import os
import logging
from pathlib import Path
from dotenv import load_dotenv

# 加载 .env 文件（显式指定路径，确保无论从哪运行都能找到）
# __file__ 是 backend/app/config.py，往上三级就是项目根目录
_PROJECT_ROOT = Path(__file__).parent.parent.parent
load_dotenv(_PROJECT_ROOT / ".env")

logger = logging.getLogger(__name__)


def _get_int(env_key: str, default: int) -> int:
    """读取整数类型的环境变量"""
    val = os.getenv(env_key, str(default))
    try:
        return int(val)
    except ValueError:
        raise ValueError(f"环境变量 {env_key} 必须是整数，当前值: {val}")


class Settings:
    """应用配置，所有配置从环境变量读取"""

    # 应用配置
    APP_HOST: str = os.getenv("APP_HOST", "0.0.0.0")
    APP_PORT: int = _get_int("APP_PORT", 8000)
    DEBUG: bool = os.getenv("DEBUG", "false").lower() in ("true", "1", "yes")

    # 数据库配置
    DATABASE_URL: str = os.getenv("DATABASE_URL", "duckdb:///./data/database.duckdb")

    # SQL 配置
    SQL_TIMEOUT: int = _get_int("SQL_TIMEOUT", 30)
    SQL_MAX_ROWS: int = _get_int("SQL_MAX_ROWS", 1000)
    SQL_MAX_RETRIES: int = _get_int("SQL_MAX_RETRIES", 3)

    # 路径配置
    BASE_DIR: Path = _PROJECT_ROOT
    DATA_DIR: Path = BASE_DIR / "data"
    LOG_DIR: Path = BASE_DIR / "logs"


# 全局配置实例
settings = Settings()


def ensure_directories():
    """确保数据目录和日志目录存在"""
    settings.DATA_DIR.mkdir(exist_ok=True)
    settings.LOG_DIR.mkdir(exist_ok=True)
```

> **逐块解释：**
>
> `_PROJECT_ROOT`：通过 `__file__`（当前文件路径）往上推三级目录，得到项目根目录。`backend/app/config.py` → `backend/app/` → `backend/` → 项目根。
>
> `load_dotenv(_PROJECT_ROOT / ".env")`：显式指定 `.env` 文件的绝对路径。这样无论你在哪个目录运行程序，都能找到 `.env`。
>
> `_get_int()`：辅助函数，把环境变量字符串转为整数。如果 `.env` 里写了 `APP_PORT=abc`，这里会报错提醒你。
>
> `Settings` 类：把所有配置集中管理。`os.getenv("KEY", "默认值")` 表示如果环境变量没设置，就用默认值。
>
> `ensure_directories()`：程序启动时调用，确保 `data/` 和 `logs/` 目录存在。

> 创建 `backend/app/db/` 目录和 `__init__.py`：

> 创建 `backend/app/db/connection.py`：

```python
import duckdb
from contextlib import contextmanager
from ..config import settings


class DatabaseConnection:
    """数据库连接管理器

    每次 get_session() 创建独立连接，
    用完自动关闭，避免并发问题。
    """

    def __init__(self, db_path: str = None):
        self.db_path = db_path or str(settings.DATA_DIR / "database.duckdb")

    def get_connection(self):
        """创建一个新的数据库连接"""
        return duckdb.connect(self.db_path)

    @contextmanager
    def get_session(self):
        """上下文管理器：自动创建和关闭连接

        用法:
            with db_connection.get_session() as conn:
                result = conn.execute("SELECT 1").fetchone()
        """
        conn = self.get_connection()
        try:
            yield conn
        finally:
            conn.close()


# 全局单例，整个应用共享
db_connection = DatabaseConnection()
```

> **关键概念：上下文管理器**
>
> `@contextmanager` 装饰器让函数可以配合 `with` 语句使用：
>
> ```python
> with db_connection.get_session() as conn:
>     # 这里 conn 是打开的连接
>     result = conn.execute("SELECT 1").fetchone()
> # 离开 with 块后，conn.close() 自动调用
> ```
>
> 即使 `execute` 抛出异常，`finally` 中的 `close()` 也会执行。这保证了连接不会泄漏。

## 2.6 编写 Schema 加载器

> Schema 加载器从数据库中读取"有哪些表、每个表有哪些列"。这是后面让 LLM 生成 SQL 的基础。

> 创建 `backend/app/db/schema_loader.py`：

```python
from typing import Dict, List, Any
from .connection import db_connection


class SchemaLoader:
    """从数据库读取表结构信息"""

    def get_tables(self) -> List[str]:
        """获取所有表名"""
        with db_connection.get_session() as conn:
            result = conn.execute("""
                SELECT table_name
                FROM information_schema.tables
                WHERE table_schema = 'main'
                ORDER BY table_name
            """).fetchall()
            return [row[0] for row in result]

    def get_table_schema(self, table_name: str) -> Dict[str, Any]:
        """获取指定表的结构（列名、类型、是否可空）"""
        with db_connection.get_session() as conn:
            columns = conn.execute("""
                SELECT column_name, data_type, is_nullable
                FROM information_schema.columns
                WHERE table_schema = 'main' AND table_name = ?
                ORDER BY ordinal_position
            """, [table_name]).fetchall()

            return {
                "table_name": table_name,
                "columns": [
                    {
                        "name": col[0],
                        "type": col[1],
                        "nullable": col[2] == "YES"
                    }
                    for col in columns
                ]
            }

    def get_full_schema(self) -> Dict[str, Any]:
        """获取所有表的完整结构"""
        tables = self.get_tables()
        schema = {}
        for table in tables:
            schema[table] = self.get_table_schema(table)
        return {"tables": schema}


# 全局单例
schema_loader = SchemaLoader()
```

> **什么是 `information_schema`？**
>
> `information_schema` 是 SQL 标准中的系统视图，存储了数据库的元数据（有哪些表、列名是什么、类型是什么）。几乎所有关系型数据库都支持它。
>
> **为什么用 `?` 占位符？**
>
> `WHERE table_name = ?` 中的 `?` 是参数占位符。数据库驱动会自动处理转义，防止 SQL 注入攻击。绝对不要用字符串拼接 SQL。

```python
from fastapi import FastAPI, HTTPException
from contextlib import asynccontextmanager

from .config import settings, ensure_directories
from .db.schema_loader import schema_loader


@asynccontextmanager
async def lifespan(app: FastAPI):
    """应用生命周期：启动时初始化"""
    ensure_directories()
    yield


app = FastAPI(
    title="Data Analyst Agent",
    lifespan=lifespan,
)


@app.get("/health")
async def health_check():
    return {"status": "healthy"}


@app.get("/api/schema")
async def get_schema():
    """获取数据库中所有表的结构"""
    try:
        schema = schema_loader.get_full_schema()
        return {"code": 200, "data": schema}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```
